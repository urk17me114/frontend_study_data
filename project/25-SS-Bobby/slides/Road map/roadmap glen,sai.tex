% Slide 13
\begin{frame}{Misuse Cases (Vulnerabilities)}
\begin{itemize}
  \item SQL Injection
  \item Unauthorized File Downloads
  \item Unrestricted File Upload
  \item Password Reset Exploits
  \item Brute-Force Attacks
  \item Typosquatting Libraries
  \item API DoS Attacks
  \item Exposed Config Files
  \item URL Access Control Bypass
  \item CAPTCHA Bypass
  \item Exposed Password Files
\end{itemize}
\end{frame}

% Slide 14
\begin{frame}{Misuse Case 1 - SQL Injection}
\textbf{Vulnerability:} Raw SQL used in login\\
\textbf{Exploit:} View DB schema, Retrieve credentials, Edit data without auth
\end{frame}

% Slide 15
\begin{frame}{Misuse Case 2 - Unauthorized File Downloads}
\textbf{Vulnerability:} No auth checks on downloads\\
\textbf{Exploit:} Brute-force filenames to access sensitive files
\end{frame}

% Slide 16
\begin{frame}{Misuse Case 3 - Unrestricted File Upload}
\textbf{Vulnerability:} Weak file upload permissions\\
\textbf{Exploit:} Upload script files without admin credentials
\end{frame}

% Slide 17
\begin{frame}{Misuse Case 4 - Password Reset Exploit}
\textbf{Vulnerability:} Weak, guessable security questions\\
\textbf{Exploit:} Reset passwords using public profile data
\end{frame}

% Slide 18
\begin{frame}{Misuse Case 5 - Brute-Force via Public Profile}
\textbf{Vulnerability:} No CAPTCHA/rate limits\\
\textbf{Exploit:} Use public data for brute-force login attempts
\end{frame}

% Slide 19
\begin{frame}{Misuse Case 6 - Typosquatting Libraries}
\textbf{Vulnerability:} Use of typo’d or outdated libraries\\
\textbf{Exploit:} Insecure dependencies in requirements.txt or backups
\end{frame}

% Slide 20
\begin{frame}{Misuse Case 7 - API Misuse for DoS}
\textbf{Vulnerability:} No API throttling\\
\textbf{Exploit:} Overload server with repeated API calls
\end{frame}

% Slide 21
\begin{frame}{Misuse Case 8 - Exposed Config Files}
\textbf{Vulnerability:} Public config file access\\
\textbf{Exploit:} Download .env, .git files for credentials
\end{frame}

% Slide 22
\begin{frame}{Misuse Case 9 - Unauthorized URL Access}
\textbf{Vulnerability:} Missing backend role checks\\
\textbf{Exploit:} Access admin URLs like /admin/logs as a student
\end{frame}

% Slide 23
\begin{frame}{Misuse Case 10 - CAPTCHA Bypass}
\textbf{Vulnerability:} Simple/bypassable CAPTCHA\\
\textbf{Exploit:} Disable JavaScript or use script bypass
\end{frame}

% Slide 24
\begin{frame}{Misuse Case 11 - Exposed Password Files}
\textbf{Vulnerability:} Poorly secured password files\\
\textbf{Exploit:} Extract admin credentials from file
\end{frame}
